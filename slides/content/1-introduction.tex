\begin{frame}{Final-Answer Accuracy Can Hide Invalid Reasoning}
  \begin{columns}[T,onlytextwidth]
    \column{0.51\textwidth}
      \sectionlabel{Corrected sodium: reference}\par
      \begin{equation}
        Na_{\mathrm{corrected}}=Na_{\mathrm{measured}}+0.024(Glucose-100)
        \label{eq:hillier-reference}
      \end{equation}
      \small Equation~\eqref{eq:hillier-reference} gives $137.248$ for $Na=127$ and $Glucose=527$.\par\vspace{0.16cm}
      \sectionlabel{Incorrect model formula}\par
      \begin{equation}
        127+0.016(527)=135.432
        \label{eq:hillier-invalid}
      \end{equation}
    \column{0.43\textwidth}
      \sectionlabel{Final-answer evaluation}\par\vspace{0.16cm}
      \begin{enumerate}
        \item Incorrect formula
        \item Numerically plausible answer
        \item Marked correct by broad tolerance
      \end{enumerate}
      \vspace{0.28cm}
      \centering
      {\large\bfseries\color{ErrorRed}Formula: Incorrect}\par\vspace{0.12cm}
      Final score accepted under the original tolerance.
  \end{columns}
  \vfill
  \centering
  {\large\bfseries\color{DeepBlue}A plausible answer is not necessarily a valid answer.}
\end{frame}
