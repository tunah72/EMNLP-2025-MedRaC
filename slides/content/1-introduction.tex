\begin{frame}{Case Study: A Correct Score from an Incorrect Formula}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \sectionlabel{1. Clinical task}\par
      A 57-year-old man has serum sodium $127$ mEq/L and glucose $527$ mg/dL. Compute corrected sodium using the Hillier equation.
      \vspace{0.22cm}

      \sectionlabel{2. Reference calculation}\par
      \begin{equation}
        \begin{aligned}
          Na_{\mathrm{corr}}
            &= Na_{\mathrm{measured}}
              + 0.024\bigl(\mathrm{Glucose}-100\bigr) \\
            &= 127 + 0.024(527-100) = 137.248
        \end{aligned}
      \end{equation}
    \column{0.48\textwidth}
      \sectionlabel{3. Model calculation}\par
      The model extracts the correct values but applies a different correction rule:
      \begin{equation}
        127+0.016(527)=135.432
      \end{equation}
      \sectionlabel{4. Evaluation outcome}\par
      Baseline score: \correctmark\;($135.432\in[130.39,144.11]$)\\
      Step-wise score: $F$ \incorrectmark; $E$ \correctmark; $C$ \correctmark; $A$ \incorrectmark
  \end{columns}
  \vspace{0.18cm}
  {\color{SeminarBlue}\rule{\linewidth}{0.7pt}}\par\vspace{0.10cm}
  \centering
  \textbf{The numerical tolerance hides a formula-selection error; the acceptable value is not a valid solution.}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Research Gap and Questions}
  \begin{columns}[T,onlytextwidth]
    \column{0.47\textwidth}
      \sectionlabel{Limitations of prior evaluation}\par\vspace{0.12cm}
      \begin{itemize}
        \item Final-answer scoring does not verify the selected medical formula.
        \item Broad numerical tolerances can accept an invalid computation path.
        \item Aggregate accuracy does not identify the first failed reasoning stage.
        \item Existing evaluations conflate knowledge, extraction, interpretation, and arithmetic errors.
      \end{itemize}
    \column{0.49\textwidth}
      \sectionlabel{Research questions}\par\vspace{0.12cm}
      \begin{enumerate}
        \item How should each stage of a medical calculation be evaluated?
        \item Which stage causes the first failure?
        \item Can formula retrieval and code execution reduce these failures?
      \end{enumerate}
  \end{columns}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Main Contributions}
  \vspace{0.08cm}
  \begin{enumerate}
    \item \textbf{Step-wise evaluation.} Evaluate each MedCalc-Bench case at Formula, Extraction, Calculation, and Answer.
    \item \textbf{Structured diagnosis.} Add Conditional Correctness, First Error Attribution, and an eight-category error taxonomy.
    \item \textbf{Training-free intervention.} Combine Formula RAG and Python execution in MedRaC; the largest reported gain is $53.19$ percentage points.
  \end{enumerate}
