\begin{frame}{Strict Correctness Requires Every Stage to Pass}
  \vspace{0.20cm}
  \centering
  \sectionlabel{Case-level criterion}\par\vspace{0.22cm}
  \begin{equation}
    \kappa=V(F)\land V(E)\land V(C)\land V(A)
    \label{eq:strict-correctness}
  \end{equation}
  \vspace{0.18cm}
  \begin{columns}[T,onlytextwidth]
    \column{0.46\textwidth}
      \sectionlabel{$V(\cdot)$}\par
      Binary validation of a stage.
    \column{0.46\textwidth}
      \sectionlabel{$\kappa$}\par
      Overall correctness for one case.
  \end{columns}
  \vfill
  \centering
  {\large\bfseries\color{DeepBlue}A correct case requires a valid Formula, Extraction, Calculation, and Answer.}
\end{frame}

\begin{frame}{Conditional Correctness Isolates Stage Reliability}
  \vspace{0.14cm}
  \centering
  \sectionlabel{Can stage $S_i$ succeed after valid prerequisites?}\par\vspace{0.20cm}
  \begin{equation}
    CC_i=P\!\left(V(S_i)\mid V(S_1)\land\cdots\land V(S_{i-1})\right)
    \label{eq:conditional-correctness}
  \end{equation}
  \vspace{0.22cm}
  \begin{columns}[T,onlytextwidth]
    \column{0.46\textwidth}
      \sectionlabel{$S_i$}\par
      The current stage in the F--E--C--A sequence.
    \column{0.46\textwidth}
      \sectionlabel{Condition}\par
      All earlier stages must already be valid.
  \end{columns}
  \vfill
  \centering
  {\large\bfseries\color{DeepBlue}Conditional Correctness separates a stage's reliability from upstream failures.}
\end{frame}

\begin{frame}{First Error Attribution Identifies the Earliest Failure}
  \vspace{0.10cm}
  \centering
  \sectionlabel{Where does an incorrect case first break?}\par\vspace{0.18cm}
  \begin{equation}
    FE_i=P\!\left(V(S_1)\land\cdots\land V(S_{i-1})\land\neg V(S_i)\mid\neg\kappa\right)
    \label{eq:first-error-attribution}
  \end{equation}
  \vspace{0.18cm}
  \begin{columns}[T,onlytextwidth]
    \column{0.46\textwidth}
      \sectionlabel{$\neg V(S_i)$}\par
      Stage $S_i$ is invalid.
    \column{0.46\textwidth}
      \sectionlabel{$\neg\kappa$}\par
      The overall case is incorrect.
  \end{columns}
  \vfill
  \centering
  {\large\bfseries\color{DeepBlue}First Error Attribution directs improvement to the first failed capability.}
\end{frame}
