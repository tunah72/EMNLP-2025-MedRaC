\begin{frame}{Formula Retrieval}
  \begin{columns}[c,onlytextwidth]
    \column{0.48\textwidth}
      \begin{enumerate}
        \item Encode the calculator question.
        \item Retrieve the most similar formula document.
        \item Condition subsequent stages on the retrieved formula.
      \end{enumerate}
    \column{0.48\textwidth}
      \centering
      {\Large
      \begin{equation}
        d^{\ast}=\arg\max_{d\in D}\operatorname{sim}(e_q,e_d)
      \end{equation}}
      {\small $e_q$: question embedding; $e_d$: formula-document embedding.}
  \end{columns}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Value Extraction and Code Generation}
  \centering
  \vspace{0.12cm}
  {\large
  \begin{equation}
    \underbrace{\text{Patient Note}}_{\text{clinical evidence}}
    +\underbrace{\text{Question}}_{\text{calculator intent}}
    +\underbrace{\text{Retrieved Formula}}_{\text{grounded method}}
    \rightarrow \text{Structured Values}
  \end{equation}}
  \vspace{0.18cm}
  {\large
  \begin{equation}
    \text{Retrieved Formula}+\text{Structured Values}+\text{Question}
    \rightarrow \text{Python Code}
  \end{equation}}
  \vspace{0.22cm}

  {\small The Patient Note supplies clinical evidence; the Question identifies the requested calculator and output.}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}[fragile]{Python-Assisted Calculation}
  \begin{columns}[c,onlytextwidth]
    \column{0.46\textwidth}
      \sectionlabel{LLM responsibility}\par\vspace{0.12cm}
      \begin{itemize}
        \item Interpret the clinical text.
        \item Identify the required patient values.
        \item Translate the grounded formula into code.
      \end{itemize}
    \column{0.50\textwidth}
      \sectionlabel{Execution responsibility}\par\vspace{0.12cm}
      {\scriptsize\begin{verbatim}
measured_na = 127
glucose = 527
result = measured_na + 0.024 * (glucose - 100)
      \end{verbatim}
      }
      {\small Python deterministically executes the generated calculation and returns the numerical result.}
  \end{columns}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}
