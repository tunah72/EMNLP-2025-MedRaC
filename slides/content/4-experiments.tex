\begin{frame}{Dataset and Benchmark Cleanup}
  \centering
  {\Large\bfseries 1,048 cases}
  \quad$\longrightarrow$\quad
  {\color{ErrorRed}\Large\bfseries 108 removed}
  \quad$\longrightarrow$\quad
  {\color{SeminarBlue}\Large\bfseries 940 retained}
  \vspace{0.38cm}

  \begin{columns}[T,onlytextwidth]
    \column{0.47\textwidth}
      \sectionlabel{Benchmark scope}\par
      \begin{itemize}
        \item 55 medical calculators sourced from MDCalc.
        \item Equation-based and rule-based cases.
        \item 940 cases retained after benchmark curation.
      \end{itemize}
    \column{0.49\textwidth}
      \sectionlabel{Representative exclusions}\par
      \begin{itemize}
        \item Mistyped gestational-age equation.
        \item Incorrect APACHE II threshold.
        \item Reversed limits for negative answers.
        \item QTc unit mismatch and invalid Caprini scoring.
      \end{itemize}
  \end{columns}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Error Reduction}
  \begin{columns}[c,onlytextwidth]

      \includegraphics[width=\linewidth,height=0.56\textheight,keepaspectratio]{paper_fig_4_llama_error_counts.png}
      \figcaption{Error counts for LLaMA3.1-8B-Instruct with and without MedRaC.}
    \column{0.27\textwidth}
      \sectionlabel{MedRaC reductions}\par\vspace{0.12cm}
      Formula\\
      $-587$ ($-77.5\%$)\\[0.18cm]
      Arithmetic\\
      $-352$ ($-82.6\%$)\\[0.18cm]

  \vspace{0.10cm}
  {\small \bluekey{Residual bottlenecks:} incorrect extraction falls by $21.3\%$; clinical misinterpretation falls by $26.9\%$. Both reductions are smaller than those for formula and arithmetic errors.}

\end{frame}

\begin{frame}{Ablations Isolate the Roles of Retrieval and Execution}
  \begin{columns}[T,onlytextwidth]
    \column{0.49\textwidth}
      \centering
      \sectionlabel{Remove Formula RAG}\par\vspace{0.10cm}
      \includegraphics[width=\linewidth,height=0.36\textheight,keepaspectratio,trim=0 30bp 0 0,clip]{paper_table_5_rag_ablation.png}
      \tabcaption{Ablation without Formula RAG.}
      \vspace{0.08cm}
      {\small
      \textbf{Accuracy:} $64.68\%\rightarrow25.64\%$ ($-39.04$ pp)\\[0.06cm]
      \textbf{Formula FE:} $20.78\%\rightarrow71.96\%$}
    \column{0.49\textwidth}
      \centering
      \sectionlabel{Remove Code Execution}\par\vspace{0.10cm}
      \includegraphics[width=\linewidth,height=0.36\textheight,keepaspectratio,trim=0 22bp 0 0,clip]{paper_table_6_code_ablation.png}
      \tabcaption{Ablation without code execution.}
      \vspace{0.08cm}
      {\small
      \textbf{Accuracy:} $64.68\%\rightarrow53.09\%$ ($-11.59$ pp)\\[0.06cm]
      \textbf{Calculation FE:} $3.23\%\rightarrow31.88\%$}
  \end{columns}
  \vspace{0.22cm}
  \centering
  {\small \bluekey{Interpretation:} retrieval primarily protects formula selection; execution primarily protects calculation.}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}
