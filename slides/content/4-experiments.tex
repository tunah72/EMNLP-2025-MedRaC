\section{Experiments and Results}

\begin{frame}{Dataset and Benchmark Cleanup}
  \centering
  {\Large\bfseries 1,048 cases}
  \quad$\longrightarrow$\quad
  {\color{ErrorRed}\Large\bfseries 108 removed}
  \quad$\longrightarrow$\quad
  {\color{SeminarBlue}\Large\bfseries 940 retained}
  \vspace{0.38cm}

  \begin{columns}[T,onlytextwidth]
    \column{0.47\textwidth}
      \sectionlabel{Benchmark scope}\par
      \begin{itemize}
        \item 55 medical calculators sourced from MDCalc.
        \item Equation-based and rule-based cases.
        \item 940 cases retained after benchmark curation.
      \end{itemize}
    \column{0.49\textwidth}
      \sectionlabel{Representative exclusions}\par
      \begin{itemize}
        \item Mistyped gestational-age equation.
        \item Incorrect APACHE II threshold.
        \item Reversed limits for negative answers.
        \item QTc unit mismatch and invalid Caprini scoring.
      \end{itemize}
  \end{columns}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\end{frame}
