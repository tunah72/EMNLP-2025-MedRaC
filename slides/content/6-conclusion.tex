\section{Discussion and Conclusion}

\begin{frame}{Scope, Limitations, and Evaluation Caveats}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.49\textwidth}
      \sectionlabel{Paper scope and evaluation caveats}\par\vspace{0.10cm}
      \begin{itemize}
        \item Structured, single-turn tasks; curated English notes, not noisy EHR or multilingual data.
        \item LLM-as-Judge; human validation: 46 cases, percent agreement only.
        \item Point estimates only; no confidence intervals or significance tests.
        \item Requires an accurate, current formula bank.
        \item Public, de-identified data; research only, not clinical deployment.
      \end{itemize}
    \column{0.47\textwidth}
      \sectionlabel{Local experiment limitations}\par\vspace{0.10cm}
      \begin{itemize}
        \item Ten fixed cases; no basis for statistical inference.
        \item Model substitution through GitHub Models and a separate 55-formula index.
        \item Post-hoc step-wise evaluation by an LLM Judge, not human annotators.
        \item One timeout repair and one answer adjudication are documented in the artifacts.
        \item Not a paper-exact reproduction.
      \end{itemize}
  \end{columns}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Conclusions}
  \vspace{0.08cm}
  \begin{enumerate}
    \item \textbf{Final-answer accuracy is insufficient for evaluating medical calculations.}
    \vspace{0.14cm}
    \item \textbf{F--E--C--A evaluation, CC, and FE identify conditional reliability and the earliest failed stage.}
    \vspace{0.14cm}
    \item \textbf{MedRaC reduces formula and arithmetic errors. Extraction and clinical interpretation remain bottlenecks.}
  \end{enumerate}
  \vspace{0.42cm}
  \centering
  {\Large\bfseries\color{DeepBlue} End-task scoring should be complemented by domain-grounded process evaluation.}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Questions and Discussion}

    \item How should an LLM-as-Judge be calibrated before it is used for high-stakes evaluation?
