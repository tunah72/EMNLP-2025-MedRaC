\begin{frame}{Appendix: Detailed Evaluation Metrics}
  \sectionlabel{Strict correctness}\par
  \small Main-deck Equation~\eqref{eq:strict-correctness} defines strict correctness.\par\vspace{0.12cm}
  \sectionlabel{Conditional Correctness}\par
  \begin{equation}
    CC_i=P\!\left(V(S_i)\mid V(S_1)\land\cdots\land V(S_{i-1})\right)
    \label{eq:app-conditional-correctness}
  \end{equation}
  \sectionlabel{First Error Attribution}\par
  \begin{equation}
    FE_i=P\!\left(V(S_1)\land\cdots\land V(S_{i-1})\land\neg V(S_i)\mid\neg\kappa\right)
    \label{eq:app-first-error-attribution}
  \end{equation}
  \sourcenote{Equations~\ref{eq:app-conditional-correctness}--\ref{eq:app-first-error-attribution}. Source: \cite{wang2026medrac}, Section 3.1.}
\end{frame}
