\begin{frame}{Appendix: Complete Metric Definitions}
  \sectionlabel{Strict correctness}\par
  \begin{equation}
    \kappa=V(F)\land V(E)\land V(C)\land V(A)
  \end{equation}
  \sectionlabel{Conditional Correctness}\par
  \begin{equation}
    CC_i=P\!\left(V(S_i)\mid V(S_1)\land\cdots\land V(S_{i-1})\right)
  \end{equation}
  \sectionlabel{First Error Attribution Rate}\par
  \begin{equation}
    FE_i=P\!\left(V(S_1)\land\cdots\land V(S_{i-1})\land\neg V(S_i)\mid\neg\kappa\right)
  \end{equation}
  \sourcenote{\cite{wang2026medrac}}
\end{frame}

\begin{frame}{Appendix: Safe-Execution Boundary}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \sectionlabel{Implemented safeguards}\par\vspace{0.10cm}
      \begin{itemize}
        \item AST node and function allowlists.
        \item No imports, filesystem, network, shell, or input.
        \item Spawned child process with a strict timeout.
        \item Result-type and magnitude checks.
      \end{itemize}
    \column{0.48\textwidth}
      \sectionlabel{Not a secure sandbox}\par\vspace{0.10cm}
      \begin{itemize}
        \item No kernel isolation.
        \item No syscall filtering.
        \item No multi-tenant security guarantee.
        \item Restricted to the seminar demonstration.
      \end{itemize}
  \end{columns}
\end{frame}
